#include<stdlib.h>

int main()
{

   while((curr=str[i++])!='\0')
   {
       switch(state){

                  if(curr=='a')
                    state=1;

                    state=3;
                  else{

                    exit(1);
                  }
                  break;
           case 1:
                  if(curr=='a')
                    state=0;
                  else if(curr=='b')
                    state=2;
                  else{
                    printf("Invalid character");
                    exit(1);
                  }
                  break;
           case 2:
                  if(curr=='a')
                    state=3;
                  else if(curr=='b')
                    state=1;
                  else{
                    printf("Ivalid character");

                  }
                  break;
           case 3:
                  if(curr=='a')
                    state=2;
                  else if(curr=='b')
                    state=0;
                  else{
                    printf("Invalid character");
                    exit(1);
                  }
                  break;
            
       }
   }
   if(state==0)
     printf("String Accepted");

}
